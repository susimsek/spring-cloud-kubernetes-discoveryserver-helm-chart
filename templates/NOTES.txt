** Please be patient while the chart is being deployed **

Spring Cloud Kubernetes Discovery Server can be accessed through the following DNS name from within your cluster:

    {{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }}.{{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }}.svc.{{ .Values.clusterDomain }} (port {{ coalesce .Values.service.ports.http .Values.service.port }})

To access Spring Cloud Kubernetes Discovery Server from outside the cluster execute the following commands:

{{- if .Values.ingress.enabled }}

1. Get the Spring Cloud Kubernetes Discovery Server URL and associate its hostname to your cluster external IP:

   export CLUSTER_IP=$(minikube ip) # On Minikube. Use: `kubectl cluster-info` on others K8s clusters
   echo "Spring Cloud Kubernetes Discovery Server URL: http{{ if .Values.ingress.tls }}s{{ end }}://{{ (tpl .Values.ingress.hostname .) }}/"
   echo "$CLUSTER_IP  {{ (tpl .Values.ingress.hostname .) }}" | sudo tee -a /etc/hosts

{{- else }}

1. Get the Spring Cloud Kubernetes Discovery Server URL by running these commands:

{{- if contains "NodePort" .Values.service.type }}

    export HTTP_NODE_PORT=$(kubectl get --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} -o jsonpath="{.spec.ports[?(@.name=='http')].nodePort}" services {{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }})
    {{- if .Values.tls.enabled }}
    export HTTPS_NODE_PORT=$(kubectl get --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} -o jsonpath="{.spec.ports[?(@.name=='https')].nodePort}" services {{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }})
    {{- end }}
    export NODE_IP=$(kubectl get nodes --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")

    echo "http://${NODE_IP}:${HTTP_NODE_PORT}/"
    {{- if .Values.tls.enabled }}
    echo "https://${NODE_IP}:${HTTPS_NODE_PORT}/"
    {{- end }}

{{- else if contains "LoadBalancer" .Values.service.type }}

  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch its status by running 'kubectl get --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} svc -w {{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }}'

    export HTTP_SERVICE_PORT=$(kubectl get --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} -o jsonpath="{.spec.ports[?(@.name=='http')].port}" services {{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }})
    {{- if .Values.tls.enabled }}
    export HTTPS_SERVICE_PORT=$(kubectl get --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} -o jsonpath="{.spec.ports[?(@.name=='https')].port}" services {{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }})
    {{- end }}
    export SERVICE_IP=$(kubectl get svc --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} {{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')

    echo "http://${SERVICE_IP}:${HTTP_SERVICE_PORT}/"
    {{- if .Values.tls.enabled }}
    echo "https://${SERVICE_IP}:${HTTPS_SERVICE_PORT}/"
    {{- end }}

{{- else if contains "ClusterIP" .Values.service.type }}

    export HTTP_SERVICE_PORT=$(kubectl get --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} -o jsonpath="{.spec.ports[?(@.name=='http')].port}" services {{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }})
    {{- if .Values.tls.enabled }}
    export HTTPS_SERVICE_PORT=$(kubectl get --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} -o jsonpath="{.spec.ports[?(@.name=='https')].port}" services {{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }})
    kubectl port-forward --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} svc/{{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }} ${HTTP_SERVICE_PORT}:${HTTP_SERVICE_PORT} ${HTTPS_SERVICE_PORT}:${HTTPS_SERVICE_PORT} &
    {{- else }}
    kubectl port-forward --namespace {{ include "spring-cloud-kubernetes-discoveryserver.namespace" . }} svc/{{ include "spring-cloud-kubernetes-discoveryserver.fullname" . }} ${HTTP_SERVICE_PORT}:${HTTP_SERVICE_PORT} &
    {{- end }}

    echo "http://127.0.0.1:${HTTP_SERVICE_PORT}/"
    {{- if .Values.tls.enabled }}
    echo "https://127.0.0.1:${HTTPS_SERVICE_PORT}/"
    {{- end }}

{{- end }}
{{- end }}
